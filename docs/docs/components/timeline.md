# Timeline

Chronological event list.

## Usage

```tsx
import { Timeline, TimelineItem, TimelineSeparator, TimelineDot, TimelineConnector, TimelineContent } from "@truongdq01/ui";

<Timeline>
  <TimelineItem>
    <TimelineSeparator>
      <TimelineDot />
      <TimelineConnector />
    </TimelineSeparator>
    <TimelineContent>Step 1</TimelineContent>
  </TimelineItem>
</Timeline>
```
